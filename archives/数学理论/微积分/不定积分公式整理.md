$$\begin{aligned}
\int x^{\mu} dx 
&= \frac{1}{\mu + 1}x^{\mu + 1} + C\ (\mu \neq -1)
\\
\int \frac{dx}{x} &= ln|x| + C
\\
\int sinx\ dx &= -cosx + C
\\
\int cosx\ dx &= sinx + C
\\
\int tanx\ dx
&= \int \frac{sinx}{cosx} dx 
= -\int \frac{d\ cosx}{cosx}
= - ln|cosx| + C
\\
\int cotx\ dx
&= \int \frac{cosx}{sinx} dx
= \int \frac{d\ sinx}{sinx}
= ln|sinx| + C
\\
\end{aligned}
$$
